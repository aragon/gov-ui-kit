export const exampleValues = ['0', '0.25', '0.5', '1', '2', '3', '4', '5', '10', '16', '32', '64', '128'];

export const DisplayComponent = ({ value }) => {
    const pxValue = Number.parseFloat(value) * 4;

    return (
        <tr>
            <td className="border-b px-4 py-2">w-{value}</td>
            <td className="border-b px-4 py-2">{`${pxValue}px`}</td>
            <td className="border-b px-4 py-2">
                <div style={{ width: pxValue }} className="bg-primary-300 h-6" />
            </td>
        </tr>
    );

};
