import { Meta } from '@storybook/addon-docs/blocks';
import classNames from 'classnames';

<Meta title="Theme/Tokens/Primitive/Border Radius" />

export const DisplayComponent = ({ className, children }) => (
    <div className={classNames('bg-neutral-0 flex h-40 w-40 items-center justify-center', className)}>{children}</div>
);

export const borderRadiuses = ['md', 'lg', 'xl', '2xl', '3xl', 'full', 'none'];

# Border Radius

<div className="flex flex-wrap p-14 gap-14 bg-neutral-50 text-neutral-800">
    {borderRadiuses.map((radius) => {
        const classes = classNames('border', {
            'rounded-md': radius === 'md',
            'rounded-lg': radius === 'lg',
            'rounded-xl': radius === 'xl',
            'rounded-2xl': radius === '2xl',
            'rounded-3xl': radius === '3xl',
            'rounded-full': radius === 'full',
            'rounded-none': radius === 'none',
        });

        return (
            <DisplayComponent key={radius} className={classes}>
                <p>{`rounded-${radius}`}</p>
            </DisplayComponent>
        );
    })}

</div>
